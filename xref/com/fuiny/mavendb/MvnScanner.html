109     @SuppressFBWarnings(value = "EI_EXPOSE_REP2", justification = "it is fine from main")
110     public void perform(Properties repos, Properties config) throws IOException {
111         this.emf = Persistence.createEntityManagerFactory(ENTITY_MANAGER_FACTORY, config);
112 
113         long start = System.currentTimeMillis();
114         this.stepRefreshIndex(
115                 repos.getProperty(REPOS_NAME),
116                 repos.getProperty(REPOS_URL));
117         this.stepScan();
118         LOG.log(Level.INFO, "Total execution time={0}", System.currentTimeMillis() - start);
119     }

121     /**
122      * Refresh the index (incremental update if not the first run).
123      *
124      * @throws IOException Exception
125      */
126     private void stepRefreshIndex(String repos, String url) throws IOException {
127 
128         // Creators we want to use (search for fields it defines)
129         List<IndexCreator> indexers = new ArrayList<>();
130         indexers.add(requireNonNull(indexCreators.get("min")));
131         indexers.add(requireNonNull(indexCreators.get("jarContent")));
132         indexers.add(requireNonNull(indexCreators.get("maven-plugin")));
133 
134         // Create context for central repository index
135         this.reposContext = indexer.createIndexingContext(
136                 repos + "-context", // ID
137                 repos, // Repository ID
138                 new File(repos + "-cache"), // Repository Directory
139                 new File(repos + "-index"), // Index Directory -  Files where local cache is (if any) and Lucene Index should be located
140                 url, null, true, true, indexers);
141 
142         Instant updateStart = Instant.now();
143         LOG.log(Level.INFO, "Refreshing Maven Index...");
144         LOG.log(Level.INFO, "This might take a while on first run, so please be patient.");
145 
146         Date centralContextCurrentTimestamp = reposContext.getTimestamp();
147         IndexUpdateRequest updateRequest = new IndexUpdateRequest(reposContext, new ResourceFetcher() {
148             private final HttpClient client = HttpClient.newBuilder().followRedirects(HttpClient.Redirect.NEVER).build();
149             private URI uri;
150 
151             @Override
152             public void connect(String id, String url) throws IOException {
153                 this.uri = URI.create(url + "/");
154             }
155 
156             @SuppressWarnings("java:S1186") // Methods should not be empty
157             @Override
158             public void disconnect() throws IOException {
159             }
160 
161             @Override
162             public InputStream retrieve(String name) throws IOException, FileNotFoundException {
163                 HttpRequest request = HttpRequest.newBuilder()
164                         .uri(uri.resolve(name))
165                         .GET()
166                         .build();
167                 try {
168                     HttpResponse<InputStream> response = client.send(request, HttpResponse.BodyHandlers.ofInputStream());
169                     if (response.statusCode() == HttpURLConnection.HTTP_OK) {
170                         return response.body();
171                     } else {
172                         throw new IOException("Unexpected response: " + response);
173                     }
174                 } catch (InterruptedException e) {
175                     Thread.currentThread().interrupt();
176                     throw new IOException(e);
177                 }
178             }
179         });
180 
181         IndexUpdateResult updateResult = indexUpdater.fetchAndUpdateIndex(updateRequest);
182         if (updateResult.isFullUpdate()) {
183             LOG.info("Full update happened!");
184         } else {
185             LOG.info(String.format("Incremental update happened, change covered %s - %s period.",
186                     centralContextCurrentTimestamp, updateResult.getTimestamp()));
187         }
188 
189         LOG.log(Level.INFO, "Finished in {0} sec", Duration.between(updateStart, Instant.now()).getSeconds());
190     }

192     /**
193      * Scan maven index files.
194      *
195      * @throws IOException Exception
196      */
197     @SuppressWarnings("java:S3776") // Cognitive Complexity of methods should not be too high
198     private void stepScan() throws IOException {
199         final IndexSearcher searcher = reposContext.acquireIndexSearcher();
200 
201         try {
202             final IndexReader ir = searcher.getIndexReader();
203             LOG.log(Level.INFO, "Maven maxDoc={0}", String.format("%,8d%n", ir.maxDoc()));
204 
205             Bits liveDocs = MultiBits.getLiveDocs(ir);
206             int i = 0;
207             for (; i < ir.maxDoc(); i++) {
208                 if (liveDocs == null || liveDocs.get(i)) {
209                     final Document doc = ir.document(i);
210                     final ArtifactInfo ai = IndexUtils.constructArtifactInfo(doc, reposContext);
211                     if (ai != null) {
212                         this.add(ai);
213                         this.store(false, i);
214                     } else {
215                         LOG.log(Level.WARNING, "{0}. ArtifactInfo is null", i);
216                     }
217                 } else {
218                     LOG.log(Level.WARNING, "{0}. This record is ignored, because of: liveDocs == null || liveDocs.get(i)", i);
219                 }
220             }
221 
222             this.store(true, i);
223         } finally {
224             reposContext.releaseIndexSearcher(searcher);
225         }
226     }

228     private void add(ArtifactInfo ai) {
229         // UInfo can never be null, based on its source code
230         //  - https://github.com/apache/maven-indexer/blob/maven-indexer-7.0.0/indexer-core/src/main/java/org/apache/maven/index/ArtifactInfo.java#L382
231         String uinfo = ai.getUinfo();
232         byte[] uinfoMd5 = DigestUtils.md5(uinfo);
233 
234         try (EntityManager em = emf.createEntityManager()) {
235             TypedQuery<Artifactinfo> query = em.createNamedQuery("Artifactinfo.findByUinfoMd5", Artifactinfo.class);
236             // Set hints for performance
237             query.setHint(QueryHints.READ_ONLY, HintValues.TRUE);
238             query.setHint(QueryHints.CACHE_STORE_MODE, CacheStoreMode.BYPASS);  // Don't insert into cache
239             query.setParameter("uinfoMd5", uinfoMd5);
240 
241             List<Artifactinfo> result = query.getResultList();
242             if (result != null && !result.isEmpty()) {
243                 LOG.log(Level.FINE, "Record exists. No update needed for {0}", ai.getUinfo());
244             } else {
245                 VersionAnalyser verAr = new VersionAnalyser(ai.getVersion());
246 
247                 // Prepare the Entity Object
248                 Artifactinfo dbAi = new Artifactinfo(uinfoMd5);
249 
250                 dbAi.setMajorVersion(verAr.getMajorVersion());
251                 dbAi.setVersionSeq(BigInteger.valueOf(verAr.getVersionSeq()));
252                 dbAi.setUinfoLength(uinfo.length());
253                 dbAi.setClassifierLength(StringUtils.length(ai.getClassifier()));
254 
255                 dbAi.setSignatureExists(ai.getSourcesExists().ordinal());
256                 dbAi.setSourcesExists(ai.getSourcesExists().ordinal());
257                 dbAi.setJavadocExists(ai.getJavadocExists().ordinal());
258 
259                 dbAi.setJson(JSON.toJson(ai));
260 
261                 // Add to DB To be saved List
262                 this.dbList.add(dbAi);
263             }
264 
265             em.clear();
266         }
267     }

269     /**
270      * Store to database.
271      *
272      * @param force Flag to force save or not
273      */
274     private void store(final boolean force, final int counter) throws IOException {
275         // Nothing to be saved
276         if (this.dbList.isEmpty()) {
277             return;
278         }
279 
280         // Save STORE_PACKAGE_SIZE records as a group,
281         // Or when force save, save it no matter of the size
282         if (this.dbList.size() >= STORE_PACKAGE_SIZE || force) {
283             try (EntityManager em = this.emf.createEntityManager()) {
284                 LocalDateTime begin = LocalDateTime.now();
285                 em.setFlushMode(FlushModeType.COMMIT);
286                 em.getTransaction().begin();
287                 this.dbList.forEach(em::persist);
288                 em.getTransaction().commit();
289                 em.clear();
290 
291                 Duration duration = Duration.between(begin, LocalDateTime.now());
292                 LOG.log(Level.INFO, "persist finished for records counter={0} in seconds={1}", new Object[]{counter, duration.toSeconds()});
293             }
294 
295             // Clear the Cached Object
296             this.dbList.clear();
297             this.dbList = new ArrayList<>(); // Add the code to avoid - java.lang.OutOfMemoryError: GC overhead limit exceeded
298         }

301     /**
302      * Version string analysis result.
303      */
304     private static final class VersionAnalyser {
305 
306         private static final long MAJOR_VERSION_MAX = 922;
307         private static final long MAJOR_VERSION_MAX_YEAR = 9223372036L;
308         private static final int RADIX_DECIMAL = 10;
309         /**
310          * Length of a year string. Example: in <code>2000.01.01</code>, so
311          * year's length is 4.
312          */
313         private static final int YEAR_LENGTH = 10;
314 
315         private static final String DOT = ".";
316         private static final String DOTS = "..";
317 
318         /**
319          * The major version of the artifact. We don't expect the major version
320          * is too big, usually it is 1, 2, 3, etc.
321          */
322         private final int majorVersionResult;
323         private final long versionSeqResult;
324 
325         /**
326          * Get maven version sequence. The value is generated by the following
327          * logic:
328          *
329          * <pre>
330          *   9,22 3,372,036,854,775,807
331          *  |- --|- ---|--- ---|--- ---|
332          *   Maj  Min  Incre   4th
333          *
334          * The left 4 digits are Major Version;
335          * then 3 digits are Minor Version;
336          * then 6 digits are Incremental Version;
337          * then the last 4 digits are the fourth Version.
338          * </pre>
339          *
340          * Well in case we suspect the version is a date format, it will be:
341          *
342          * <pre>
343          *   9,223,372,036,854,775,807
344          *   - --- --- ---|--- ---|---|
345          *            Year   Month Date
346          * Example version text
347          * - com.hack23.cia           | citizen-intelligence-agency | 2016.12.13
348          * - org.everit.osgi.dev.dist | eosgi-dist-felix_5.2.0      | v201604220928
349          * - berkano                  | berkano-sample              | 20050805.042414
350          * </pre>
351          *
352          * @param version Version string
353          */
354         @SuppressWarnings({"checkstyle:MagicNumber", "java:S3776"}) // java:S3776 - Cognitive Complexity of methods should not be too high
355         private VersionAnalyser(final String version) {
356             String majorVersionStr = "";
357             long majorVersion = 0;
358             long minorVersion = 0;
359             long increVersion = 0;
360             long four4Version = 0;
361 
362             String versionTemp = version.replaceAll("[^\\d.]", DOT);
363             if (versionTemp.contains(DOT)) {
364                 while (versionTemp.contains(DOTS)) {
365                     versionTemp = versionTemp.replace(DOTS, DOT);
366                 }
367                 StringTokenizer tok = new StringTokenizer(versionTemp, DOT);
368 
369                 if (tok.hasMoreTokens()) {
370                     majorVersionStr = tok.nextToken();
371                     majorVersion = NumberUtils.toLong(majorVersionStr);
372                     majorVersion = (majorVersion > VersionAnalyser.MAJOR_VERSION_MAX) ? VersionAnalyser.MAJOR_VERSION_MAX : majorVersion;
373                 }
374                 if (tok.hasMoreTokens()) {
375                     minorVersion = NumberUtils.toLong(tok.nextToken());
376                 }
377                 if (tok.hasMoreTokens()) {
378                     increVersion = NumberUtils.toLong(tok.nextToken());
379                 }
380                 if (tok.hasMoreTokens()) {
381                     four4Version = NumberUtils.toLong(tok.nextToken());
382                 }
383             } else {
384                 four4Version = NumberUtils.toLong(versionTemp);
385             }
386 
387             long seq;
388             if (majorVersion == VersionAnalyser.MAJOR_VERSION_MAX) {
389                 // We suspect the version string is usually a year.month.date
390                 // So we set major version as the year
391                 String upTo4char = majorVersionStr.substring(0, Math.min(majorVersionStr.length(), YEAR_LENGTH));
392                 majorVersion = NumberUtils.toLong(upTo4char);
393                 seq = shrinkLong(NumberUtils.toLong(majorVersionStr),
394                         MAJOR_VERSION_MAX_YEAR) * 1000000000L + shrinkLong(minorVersion, 999999) * 1000L + shrinkLong(increVersion, 999);
395             } else {
396                 // All other cases
397                 seq = majorVersion * 10000000000000000L
398                         + shrinkLong(minorVersion, 9999) * 1000000000000L
399                         + shrinkLong(increVersion, 999999) * 1000000L
400                         + shrinkLong(four4Version, 999999);
401             }
402 
403             // Set results
404             this.majorVersionResult = (majorVersion >= Integer.MAX_VALUE) ? Integer.MAX_VALUE : (int) majorVersion;
405             this.versionSeqResult = seq;
406         }
407 
408         /**
409          * Return {@link #majorVersionResult} value.
410          *
411          * @return Value of {@link #majorVersionResult}

413         int getMajorVersion() {

418          * Return {@link #versionSeqResult} value.

422         long getVersionSeq() {

426         /**
427          * {@inheritDoc}
428          */
429         @Override
430         public String toString() {

433 
434         /**
435          * Shrink a long value to avoid it exceed the limit.
436          *
437          * @param value Value to shrink
438          * @param limit Max value allowed
439          * @return Value no more than the limit
440          */
441         private long shrinkLong(final long value, final long limit) {
442             long temp = value;
443             while (temp > limit) {
444                 temp = temp / RADIX_DECIMAL;
445             }
446 
447             return temp;
448         }
